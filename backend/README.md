# Instalação
    > pip install -r requirements.txt

# Executar sem debug
    > flask run

# Executar com debug
    > flask --debug run 

# Executar testes unitários e de integração 
    > py.test 

# Executar executar os testes e receber um relatório da porcentagem de código coberta com testes
    > py.test --cov=app
